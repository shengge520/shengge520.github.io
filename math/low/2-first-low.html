# 第二题：找出下列正数组的最大差值

输入 [10,5,11,7,8,9] 输出 6

  • 方法一：
function getGapValue(arr){
    var max=arr[0];
    var min=arr[0];
    var res=0;
    for (var i = 1; i < arr.length; i++) {
        if(arr[i]>max){
            max=arr[i];
        }
        if(arr[i]<min){
            min=arr[i]
        }
        res=max-min;
    }
    return res;
}
  • 方法二：
function getGapValue(arr){
    const item1 = Math.max.apply(null,arr);
    const item2 = Math.min.apply(null,arr);
    return item1 - item2; 
}
